  <head>

    <meta name="viewport" content="width=device-width, initial-scale=1" />

    <script src="https://aframe.io/releases/1.4.2/aframe.min.js"></script>

    <script src="https://cdn.jsdelivr.net/npm/mind-ar@1.2.2/dist/mindar-image-aframe.prod.js"></script>

  </head>

  <body>

    

    <div style="position: absolute; z-index: 1000; bottom: 20px; left: 50%; transform: translateX(-50%);">

        <button id="startButton" style="padding: 10px 20px; font-size: 18px;">เริ่มใช้งาน (Start AR)</button>

    </div>


    <a-scene mindar-image="imageTargetSrc: ./targets.mind;" color-space="sRGB" renderer="colorManagement: true, physicallyCorrectLights" vr-mode-ui="enabled: false" device-orientation-permission-ui="enabled: false">

      

      <a-assets>

        <audio id="myAudio" src="your-audio-file.mp3" preload="auto"></audio>

      </a-assets>


      <a-camera position="0 0 0" look-controls="enabled: false"></a-camera>


      <a-entity id="example-target" mindar-image-target="targetIndex: 0">

        

        <a-text value="Hello AR World!" color="yellow" align="center" width="2" position="0 0.5 0"></a-text>

        

        <a-plane src="#myImage" position="0 0 0" height="0.5" width="0.5" rotation="0 0 0"></a-plane>


      </a-entity>

    </a-scene>


    <script>

      const startButton = document.querySelector('#startButton');

      const audio = document.querySelector('#myAudio');

      const targetEntity = document.querySelector('#example-target');


      // เมื่อกดปุ่ม ให้ปลดล็อกเสียงและเริ่มระบบ

      startButton.addEventListener('click', () => {

        startButton.style.display = 'none';

        // เล่นเสียงทิ้งไว้เบาๆ หรือ Pause ไว้ก่อนเพื่อรอ Trigger

        audio.play().then(() => { audio.pause(); }); 

      });


      // ดักจับเหตุการณ์เมื่อสแกนเจอรูป (Target Found)

      targetEntity.addEventListener("targetFound", event => {

        console.log("เจอรูปแล้ว!");

        audio.play(); // เล่นเสียงทันที

      });


      // ดักจับเหตุการณ์เมื่อเอารูปออก (Target Lost)

      targetEntity.addEventListener("targetLost", event => {

        console.log("รูปหายไปแล้ว");

        audio.pause(); // หยุดเสียง (หรือจะปล่อยให้เล่นต่อก็ได้)

        audio.currentTime = 0; // รีเซ็ตเสียงไปเริ่มต้นใหม่

      });

    </script>
